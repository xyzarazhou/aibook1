import React, { useState, useEffect, useRef, useMemo } from 'react';
import { Play, Pause, SkipForward, SkipBack, Volume2, Globe, BookOpen, Sparkles, Image as ImageIcon, Loader2 } from 'lucide-react';

// --- Configuration & Constants ---
const API_KEY = ""; // Provided by environment
const TTS_MODEL = "gemini-2.5-flash-preview-tts";
const IMAGE_MODEL = "imagen-4.0-generate-001";
const DEFAULT_VOICE = "Kore"; 

const SAMPLE_STORY = {
  title: "小兔子的花园",
  pinyinTitle: "Xiǎo tù zi de huā yuán",
  englishTitle: "The Little Rabbit's Garden",
  pages: [
    {
      text: "春天来了，小兔子在花园里种花。",
      // Corrected: Each pinyin entry matches exactly one word entry
      words: ["春天", "来了", "，", "小兔子", "在", "花园", "里", "种", "花", "。"],
      pinyin: ["chūn tiān", "lái le", "，", "xiǎo tù zi", "zài", "huā yuán", "lǐ", "zhòng", "huā", "。"],
      translation: "Spring is here, the little rabbit is planting flowers in the garden.",
      imagePrompt: "A cute fluffy white rabbit wearing a small sun hat, planting colorful tulips in a sunny green garden, soft watercolor style, high quality children's book illustration."
    },
    {
      text: "它每天给小花浇水，小花长得很快。",
      words: ["它", "每天", "给", "小花", "浇水", "，", "小花", "长得", "很快", "。"],
      pinyin: ["tā", "měi tiān", "gěi", "xiǎo huā", "jiāo shuǐ", "，", "xiǎo huā", "zhǎng de", "hěn kuài", "。"],
      translation: "It waters the flowers every day, and the flowers grow very quickly.",
      imagePrompt: "The rabbit holding a small blue watering can, watering bright red and yellow flowers, sparkling water drops, cheerful atmosphere, soft watercolor style."
    },
    {
      text: "蝴蝶飞来了，花园变得真漂亮！",
      words: ["蝴蝶", "飞来了", "，", "花园", "变得", "真", "漂亮", "！"],
      pinyin: ["hú dié", "fēi lái le", "，", "huā yuán", "biàn de", "zhēn", "piào liang", "！"],
      translation: "Butterflies flew over, the garden has become so beautiful!",
      imagePrompt: "Many colorful butterflies flying around a blooming garden, a happy rabbit looking up at them, vibrant colors, soft watercolor style."
    }
  ]
};

// Helper for exponential backoff retries
const fetchWithRetry = async (url, options, maxRetries = 5) => {
  let delay = 1000;
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(url, options);
      if (response.ok) return response;
      if (response.status === 429 || response.status >= 500) {
        // Retryable errors - wait and retry
      } else {
        throw new Error(`Request failed with status ${response.status}`);
      }
    } catch (err) {
      if (i === maxRetries - 1) throw err;
    }
    await new Promise(resolve => setTimeout(resolve, delay));
    delay *= 2;
  }
  throw new Error('Max retries reached');
};

const App = () => {
  const [currentPage, setCurrentPage] = useState(0);
  const [isPlaying, setIsPlaying] = useState(false);
  const [activeWordIndex, setActiveWordIndex] = useState(-1);
  const [audioUrl, setAudioUrl] = useState(null);
  const [pageImages, setPageImages] = useState({});
  const [isAudioLoading, setIsAudioLoading] = useState(false);
  const [isImageLoading, setIsImageLoading] = useState(false);
  const [showPinyin, setShowPinyin] = useState(true);
  const [showTranslation, setShowTranslation] = useState(true);
  const [error, setError] = useState(null);
  
  const audioRef = useRef(null);
  const pageData = SAMPLE_STORY.pages[currentPage];

  // --- Image Generation Logic ---
  const fetchImage = async (prompt, pageIndex) => {
    if (pageImages[pageIndex]) return; 
    
    setIsImageLoading(true);
    try {
      const response = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/${IMAGE_MODEL}:predict?key=${API_KEY}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          instances: { prompt: prompt },
          parameters: { sampleCount: 1 }
        })
      });

      const result = await response.json();
      if (!result.predictions?.[0]?.bytesBase64Encoded) throw new Error('Invalid image data');
      
      const imageUrl = `data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`;
      setPageImages(prev => ({ ...prev, [pageIndex]: imageUrl }));
    } catch (err) {
      console.error("Image error:", err);
    } finally {
      setIsImageLoading(false);
    }
  };

  // --- Audio Generation Logic ---
  const fetchAudio = async (text) => {
    setIsAudioLoading(true);
    setError(null);
    try {
      const response = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/${TTS_MODEL}:generateContent?key=${API_KEY}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: `Say gently: ${text}` }] }],
          generationConfig: {
            responseModalities: ["AUDIO"],
            speechConfig: {
              voiceConfig: { prebuiltVoiceConfig: { voiceName: DEFAULT_VOICE } }
            }
          }
        })
      });

      const data = await response.json();
      const pcmBase64 = data.candidates?.[0]?.content?.parts?.[0]?.inlineData?.data;
      
      if (!pcmBase64) throw new Error('No audio data received');

      const wavBlob = pcmToWav(pcmBase64, 24000);
      const url = URL.createObjectURL(wavBlob);
      setAudioUrl(url);
    } catch (err) {
      setError("Unable to load audio. Please check connection.");
      console.error(err);
    } finally {
      setIsAudioLoading(false);
    }
  };

  const pcmToWav = (base64, sampleRate) => {
    const binaryString = atob(base64);
    const len = binaryString.length;
    const bytes = new Uint8Array(len);
    for (let i = 0; i < len; i++) bytes[i] = binaryString.charCodeAt(i);
    
    const buffer = new ArrayBuffer(44 + len);
    const view = new DataView(buffer);
    const writeString = (offset, string) => {
      for (let i = 0; i < string.length; i++) view.setUint8(offset + i, string.charCodeAt(i));
    };

    writeString(0, 'RIFF');
    view.setUint32(4, 32 + len, true);
    writeString(8, 'WAVE');
    writeString(12, 'fmt ');
    view.setUint32(16, 16, true);
    view.setUint16(20, 1, true);
    view.setUint16(22, 1, true);
    view.setUint32(24, sampleRate, true);
    view.setUint32(28, sampleRate * 2, true);
    view.setUint16(32, 2, true);
    view.setUint16(34, 16, true);
    writeString(36, 'data');
    view.setUint32(40, len, true);

    for (let i = 0; i < len; i++) view.setUint8(44 + i, bytes[i]);
    return new Blob([buffer], { type: 'audio/wav' });
  };

  useEffect(() => {
    setAudioUrl(null);
    setIsPlaying(false);
    setActiveWordIndex(-1);
    fetchAudio(pageData.text);
    fetchImage(pageData.imagePrompt, currentPage);
  }, [currentPage]);

  useEffect(() => {
    let interval;
    if (isPlaying && audioRef.current) {
      interval = setInterval(() => {
        const currentTime = audioRef.current.currentTime;
        const duration = audioRef.current.duration || 1;
        const progress = currentTime / duration;
        const totalWords = pageData.words.length;
        const index = Math.floor(progress * totalWords);
        
        // Progressively turn characters green
        if (index < totalWords && index > activeWordIndex) {
          setActiveWordIndex(index);
        }
      }, 100);
    } else {
      clearInterval(interval);
    }
    return () => clearInterval(interval);
  }, [isPlaying, pageData.words.length, activeWordIndex]);

  const togglePlay = () => {
    if (!audioRef.current) return;
    if (isPlaying) {
      audioRef.current.pause();
    } else {
      audioRef.current.play();
    }
    setIsPlaying(!isPlaying);
  };

  return (
    <div className="min-h-screen bg-orange-50 font-sans text-gray-800 p-4 md:p-8 flex flex-col items-center">
      <header className="w-full max-w-5xl flex justify-between items-center mb-8 bg-white p-4 rounded-2xl shadow-sm border-b-2 border-orange-100">
        <div className="flex items-center gap-3">
          <div className="bg-orange-400 p-2 rounded-lg">
            <BookOpen className="text-white w-6 h-6" />
          </div>
          <div>
            <h1 className="text-xl font-bold text-orange-900">{SAMPLE_STORY.title}</h1>
            <p className="text-xs text-orange-600 font-medium tracking-wider uppercase">{SAMPLE_STORY.englishTitle}</p>
          </div>
        </div>
        
        <div className="flex gap-2">
          <button 
            onClick={() => setShowPinyin(!showPinyin)}
            className={`px-3 py-1.5 rounded-full text-sm font-bold transition-all ${showPinyin ? 'bg-green-100 text-green-700 ring-2 ring-green-200' : 'bg-gray-100 text-gray-500'}`}
          >
            拼音 Pinyin
          </button>
          <button 
            onClick={() => setShowTranslation(!showTranslation)}
            className={`px-3 py-1.5 rounded-full text-sm font-bold transition-all ${showTranslation ? 'bg-blue-100 text-blue-700 ring-2 ring-blue-200' : 'bg-gray-100 text-gray-500'}`}
          >
            EN
          </button>
        </div>
      </header>

      <main className="w-full max-w-5xl grid grid-cols-1 lg:grid-cols-2 gap-8 bg-white rounded-3xl p-6 lg:p-10 shadow-xl border-4 border-orange-200">
        
        {/* Left: Illustration */}
        <div className="relative aspect-square bg-gray-100 rounded-2xl overflow-hidden shadow-inner border border-gray-100">
          {isImageLoading && !pageImages[currentPage] ? (
            <div className="absolute inset-0 flex flex-col items-center justify-center gap-4 bg-gray-50">
              <Loader2 className="w-10 h-10 text-orange-400 animate-spin" />
              <p className="text-orange-900 font-medium text-sm animate-pulse">Drawing page {currentPage + 1}...</p>
            </div>
          ) : pageImages[currentPage] ? (
            <img 
              src={pageImages[currentPage]} 
              alt="Story illustration" 
              className="w-full h-full object-cover animate-in fade-in zoom-in duration-700"
            />
          ) : (
            <div className="absolute inset-0 flex flex-col items-center justify-center gap-4">
              <ImageIcon className="w-16 h-16 text-gray-300" />
              <p className="text-gray-400 text-sm">Illustration pending...</p>
            </div>
          )}
        </div>

        {/* Right: Text & Controls */}
        <div className="flex flex-col justify-between py-2">
          <div className="space-y-8">
            <div className="flex flex-wrap gap-x-3 gap-y-8 leading-relaxed items-end min-h-[180px] px-2">
              {pageData.words.map((word, idx) => (
                <div key={idx} className="flex flex-col items-center group">
                  {showPinyin && (
                    <span className={`text-sm md:text-lg font-bold font-mono mb-1 transition-colors duration-300 ${
                      idx <= activeWordIndex ? 'text-green-600' : 'text-gray-300'
                    }`}>
                      {pageData.pinyin[idx]}
                    </span>
                  )}
                  <span 
                    className={`text-4xl md:text-6xl font-bold transition-all duration-300 ${
                      idx <= activeWordIndex 
                        ? 'text-green-600' 
                        : 'text-gray-800'
                    }`}
                  >
                    {word}
                  </span>
                </div>
              ))}
            </div>

            {showTranslation && (
              <div className="p-6 bg-blue-50 rounded-2xl border-l-8 border-blue-400 animate-in fade-in slide-in-from-left-4">
                <p className="text-blue-900 text-xl md:text-2xl font-medium italic leading-snug">
                  "{pageData.translation}"
                </p>
              </div>
            )}
          </div>

          <div className="mt-12 space-y-8">
            {error && <div className="p-3 bg-red-50 text-red-600 rounded-xl text-center font-medium text-sm">{error}</div>}
            
            <div className="flex items-center gap-8">
              <button 
                onClick={() => setCurrentPage(Math.max(0, currentPage - 1))}
                disabled={currentPage === 0}
                className="p-4 bg-gray-100 hover:bg-orange-100 disabled:opacity-30 rounded-2xl transition-all active:scale-95 group"
              >
                <SkipBack className="w-8 h-8 text-gray-600 group-hover:text-orange-600" />
              </button>

              <button 
                onClick={togglePlay}
                disabled={isAudioLoading || !audioUrl}
                className={`w-20 h-20 flex items-center justify-center rounded-full shadow-lg transition-all active:scale-90 ${
                  isAudioLoading ? 'bg-gray-300' : 'bg-orange-500 hover:bg-orange-600'
                }`}
              >
                {isAudioLoading ? (
                  <Loader2 className="w-10 h-10 text-white animate-spin" />
                ) : isPlaying ? (
                  <Pause className="w-10 h-10 text-white fill-current" />
                ) : (
                  <Play className="w-10 h-10 text-white fill-current ml-1" />
                )}
              </button>

              <button 
                onClick={() => setCurrentPage(Math.min(SAMPLE_STORY.pages.length - 1, currentPage + 1))}
                disabled={currentPage === SAMPLE_STORY.pages.length - 1}
                className="p-4 bg-gray-100 hover:bg-orange-100 disabled:opacity-30 rounded-2xl transition-all active:scale-95 group"
              >
                <SkipForward className="w-8 h-8 text-gray-600 group-hover:text-orange-600" />
              </button>
              
              <div className="flex-1 flex flex-col gap-2">
                <div className="flex justify-between text-xs font-bold text-orange-400 uppercase tracking-widest">
                  <span>Page Progress</span>
                  <span>{currentPage + 1} / {SAMPLE_STORY.pages.length}</span>
                </div>
                <div className="h-3 bg-gray-100 rounded-full overflow-hidden p-0.5">
                  <div 
                    className="h-full bg-orange-400 rounded-full transition-all duration-700 ease-out"
                    style={{ width: `${((currentPage + 1) / SAMPLE_STORY.pages.length) * 100}%` }}
                  ></div>
                </div>
              </div>
            </div>
          </div>
        </div>
      </main>

      <footer className="mt-12 text-gray-400 text-sm flex items-center gap-8">
        <div className="flex items-center gap-2">
          <Volume2 className="w-4 h-4" />
          <span>Gemini AI Narrator (Kore)</span>
        </div>
        <div className="flex items-center gap-2">
          <Sparkles className="w-4 h-4" />
          <span>Imagen AI Illustrations</span>
        </div>
      </footer>

      <style dangerouslySetInnerHTML={{ __html: `
        @keyframes fadeIn {
          from { opacity: 0; transform: translateY(10px); }
          to { opacity: 1; transform: translateY(0); }
        }
        .animate-in { animation: fadeIn 0.5s ease-out forwards; }
      `}} />
    </div>
  );
};

export default App;
